075            public void testFromFull(final Class<? extends Operation<Statements.TestFull, Statements.TestFull>> opClass) {
076                    for(int i = 0; i <= ValuesCount; ++i) {
077                            final int valIdx = i;
078                            final Statements.TestFull testFull;
079                            
080                            if (valIdx < ValuesCount) {
081                                    testFull = new Statements.TestFull() {
082                                            @Override public Long getId() { return (long)(valIdx + 1); }
083                                            @Override public String getStr() { return ValuesStr[valIdx]; }
084                                            @Override public BigDecimal getDez() { return ValuesDez[valIdx]; }
085                                            @Override public java.sql.Date getDay() { return new java.sql.Date(ValuesDay[valIdx].getTime()); }
086                                            @Override public Integer getIdx() { return ValuesIdx[valIdx]; }
087                                            @Override public Class<?> getNoSuchColumn() { return null; }
088                                            @Override public String getDescription() { return null; }
089                                    };
090                            }
091                            else {
092                                    testFull = new Statements.TestFull() {
093                                            @Override public Long getId() { return (long)(valIdx + 1); }
094                                            @Override public String getStr() { return null; }
095                                            @Override public BigDecimal getDez() { return null; }
096                                            @Override public java.sql.Date getDay() { return null; }
097                                            @Override public Integer getIdx() { return null; }
098                                            @Override public Class<?> getNoSuchColumn() { return null; }
099                                            @Override public String getDescription() { return null; }
100                                    };
101                            }
102                            
103                            /* Run through DB */
104                            Statements.TestFull testFullOut = executor.get(Statements.TestFromFull.class, testFull);
105                            Assert.assertNotNull("One row expected, none found (valIdx=" + valIdx + ")", testFullOut);
106                            Assert.assertEquals(testFullOut, testFullOut);
107                            Assert.assertEquals(testFullOut.hashCode(), testFullOut.hashCode());
108                            
109                            /* Compare fields */
110                            Assert.assertEquals(testFull.getId(), testFullOut.getId());
111                            Assert.assertEquals(testFull.getStr(), testFullOut.getStr());
112                            if ((testFull.getDez() != null) && (testFullOut.getDez() != null))
113                                    Assert.assertEquals(testFull.getDez().stripTrailingZeros(), testFullOut.getDez().stripTrailingZeros());
114                            else
115                                    Assert.assertEquals(testFull.getDez(), testFullOut.getDez());
116                            Assert.assertEquals(testFull.getDay(), testFullOut.getDay());
117                            Assert.assertEquals(testFull.getIdx(), testFullOut.getIdx());
118                            Assert.assertEquals((new Statements()).getDescription(), testFullOut.getDescription());
119                            Assert.assertEquals(testFull.getNoSuchColumn(), testFullOut.getNoSuchColumn());
120                            
121                            /* Validate toString() */
122                            Assert.assertEquals(
123                                    "{" +
124                                            "getId(): " + testFullOut.getId() + "; " +
125                                            "getStr(): " + testFullOut.getStr() + "; " +
126                                            "getDez(): " + testFullOut.getDez() + "; " +
127                                            "getDay(): " + testFullOut.getDay() + "; " +
128                                            "getIdx(): " + testFullOut.getIdx() + "; " +
129                                            "getDescription(): " + (new Statements()).getDescription() +
130                                    "}",
131                                    testFullOut.toString()
132                            );
133                    }
134            }
