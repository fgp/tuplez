018    /**
019     * Executes concrete {@link Operation}s.
020     * <p>
021     * Executor instances are created from {@link DataSource}s,
022     * and contain instance methods which execute concrete
023     * {@link Operation} implementations.
024     * <p>
025     * Executor instances or typically created by Spring.
026     * This is e.g. done by putting the following into your
027     * application context configuration XML (assuming that
028     * there is a bean named "DataSource" which defines
029     * your database connection).
030     *<blockquote><pre>{@literal
031     *<bean id="DatabaseExecutor"
032     *      class="org.phlo.tuplez.Executor"
033     *>
034     *    <property name="DataSource" ref="DataSource"/>
035     *</bean>
036     *}</pre></blockquote>
037     *
038     * @see Operation
039     */
040    public class Executor {
041            private static final SqlParameterSource s_noDefaultInputSqlParameterSource = new SqlParameterSource() {
042                    @Override public boolean hasValue(String paramName) { return false; }
043                    @Override public Object getValue(String paramName) { throw new IllegalArgumentException("No default input configured"); }
044                    @Override public int getSqlType(String paramName) { return TYPE_UNKNOWN; }
045                    @Override public String getTypeName(String paramName) { return null; }
046            };

048            /* The named-parameter JDBC template used to execute operations */
049            private NamedParameterJdbcTemplate m_npJdbcTemplate;

051            /* The parameter source for default.* input parameters */
052            private SqlParameterSource m_defaultInputSqlParameterSource = s_noDefaultInputSqlParameterSource;

117                    if (defaultInputSqlParameterSource == null)

146            /**
147             * Executes a database operation using the provided
148             * instance of {@code InputType} to fill in parameters,
149             * passes an iterator over the resulting instances of
150             * {@code OutputType} to the {@code iteratorProcessor},
151             * and returns the {@code iteratorProcessor}'s result
152             * of type {@code ResultType}.
153             * <p>
154             * If the database operation requires no input (meaning
155             * {@code InputType} is {@link Void}) you must use
156             * {@link #iterate(Class, IteratorProcessor) iterate(opClass, iteratorProcessor)}
157             * instead, since {@link Void} is uninstantiable.
158             * <p>
159             * It's usually easier to use
160             * {@link #collection(Class, Object) collection(opClass, input)}
161             * instead of using this method directly
162             * 
163             * @see Operation
164             * @see #iterate(Class opClass, IteratorProcessor iteratorProcessor)
165             * @see #collection(Class opClass, Object input)
166             * @see #collection(Class opClass)
167             * 
168             * @param <InputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s input type
169             * @param <ConcreteInputType> the actual (concrete) implementation of {@code InputType}
170             * @param <OpClassType> the operation class'es actual type, extends {@link Operation Operation&lt;InputType, OutputType&gt;}
171             * @param <OutputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s output type
172             *                      and {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;}'s element type
173             * @param <ResultType> the {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;}'s result type
174             * @param opClass the concrete {@link Operation Operation&lt;InputType, OutputType&gt;} implementation
175             * @param input the {@code ConcreteInputType} instance used to fill in the operation's parameters
176             * @param iteratorProcessor the {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;} instance
177             * @return the {@code iteratorProcessor}'s result of type {@code ResultType}
178             */
179            public <
180                    InputType,
181                    ConcreteInputType extends InputType, OutputType,
182                    OpClassType extends Operation<InputType, OutputType>,
183                    ResultType
184            >
185            ResultType iterate(
186                    final Class<OpClassType> opClass,
187                    final ConcreteInputType input,
188                    final IteratorProcessor<OutputType, ResultType> iteratorProcessor
189            ) {
190                    return m_npJdbcTemplate.query(
191                            StatementMapper.getInstance(opClass).getStatement(input),
192                            InputMapper.getInstance(opClass).mapInput(this, input),
193                            new ResultSetExtractor<ResultType>() {
194                                    public ResultType extractData(final ResultSet resultSet) throws SQLException
195                                    {
196                                            try {
197                                                    return iteratorProcessor.processIterator(
198                                                            new ResultSetIterator<OutputType>(
199                                                                    resultSet,
200                                                                    ResultSetMapper.getInstance(opClass, resultSet)
201                                                            )
202                                                    );
203                                            }
204                                            catch (WrappedSQLException e) {
205                                                    throw e.getSQLException();
206                                            }
207                                    }
208                                    
209                            }
210                    );
211            }

213            /**
214             * Executes a database operation which requires no input,
215             * passes an iterator over the resulting instances of
216             * {@code OutputType} to the {@code iteratorProcessor},
217             * and returns the {@code iteratorProcessor}'s result
218             * of type {@code ResultType}.
219             * <p>
220             * If the database operation requires input (meaning
221             * {@code InputType} is not {@link Void}) you must use
222             * {@link #iterate(Class, Object, IteratorProcessor) iterate(opClass, input, iteratorProcessor)}
223             * instead.
224             * <p>
225             * It's usually easier to use the member
226             * {@link #collection(Class) collection(opClass)} instead
227             * of using this method directly.
228             * 
229             * @see Operation
230             * @see #iterate(Class opClass, Object input, IteratorProcessor iteratorProcessor)
231             * @see #collection(Class opClass, Object input)
232             * @see #collection(Class opClass)
233             * 
234             * @param <OutputType> the {@link Operation Operation&lt;Void, OutputType&gt;}'s output type
235             *                      and {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;}'s element type
236             * @param <OpClassType> the operation class'es actual type, extends {@link Operation Operation&lt;Void, OutputType&gt;}
237             * @param <ResultType> the {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;}'s result type
238             * @param opClass the concrete {@link Operation Operation&lt;Void, OutputType&gt;} implementation
239             * @param iteratorProcessor the {@link IteratorProcessor IteratorProcessor&lt;OutputType, ResultType&gt;} instance
240             * @return the {@code iteratorProcessor}'s result of type {@code ResultType}
241             */
242            public <
243                    OutputType,
244                    OpClassType extends Operation<Void, OutputType>,
245                    ResultType
246            >
247            ResultType iterate(
248                    final Class<OpClassType> opClass,
249                    final IteratorProcessor<OutputType, ResultType> iteratorProcessor
250            ) {
251                    return iterate(opClass, null, iteratorProcessor);
252            }

254            /**
255             * Executes a database operation using the provided
256             * instance of {@code InputType} to fill in parameters,
257             * and returns a {@link Collection} of the resulting
258             * instances of {@code OutputType}.
259             * <p>
260             * If the database operation requires no input (meaning
261             * {@code <InputType} is {@link Void}) you must use
262             * {@link #collection(Class) collection(opClass)} instead,
263             * since {@link Void} is uninstantiable.
264             * 
265             * @see Operation
266             * @see #collection(Class)
267             * 
268             * @param <InputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s input type
269             * @param <ConcreteInputType> the actual (concrete) implementation of {@code InputType}
270             * @param <OpClassType> the operation class'es actual type, extends {@link Operation Operation&lt;InputType, OutputType&gt;}
271             * @param <OutputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s output type
272             * @param opClass the concrete {@link Operation Operation&lt;InputType, OutputType&gt;} implementation
273             * @param input the {@code ConcreteInputType} instance used to fill in the operation's parameters
274             * @return the resulting {@link Collection Collection&lt;OutputType&gt;} of {@code OutputType} instances
275             */
276            public <
277                    InputType,
278                    ConcreteInputType extends InputType,
279                    OutputType,
280                    OpClassType extends Operation<InputType, OutputType>
281            >
282            Collection<OutputType> collection(
283                    final Class<OpClassType> opClass,
284                    final ConcreteInputType input
285            ) {
286                    return iterate(opClass, input, new IteratorProcessor<OutputType, Collection<OutputType>>() {
287                            public Collection<OutputType> processIterator(Iterator<OutputType> iterator) {
288                                    Collection<OutputType> collection = new java.util.LinkedList<OutputType>();
289                                    while (iterator.hasNext())
290                                            collection.add(iterator.next());
291                                    return collection;
292                            }
293                            
294                    });
295            }

325            /**
326             * Executes a single-row-returning database operation
327             * (flagged as such by implementing the interface
328             * {@link ReturnsSingleRow}) using the
329             * provided instance of {@code InputType} to fill in
330             * parameters, and returns a single {@code OutputType}
331             * instance representing the single result row, or
332             * {@code null} if no row was produced.
333             * <p>
334             * It is an error for an operation executed with
335             * this method to return more than one row. Doing
336             * so triggers an {@link InvalidDataAccessApiUsageException}
337             * <p>
338             * If the database operation requires no input (meaning
339             * {@code <InputType} is {@link Void}) you must use
340             * {@link #get(Class) get(opClass)}
341             * instead, since {@link Void} is uninstantiable.
342             * 
343             * @see ReturnsSingleRow
344             * @see #get(Class opClass)
345             * 
346             * @param <InputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s input type
347             * @param <ConcreteInputType> the actual (concrete) implementation of {@code InputType}
348             * @param <OutputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s output type
349             * @param <OpClassType> the operation class'es actual type, extends {@link Operation Operation&lt;InputType, OutputType&gt;}
350             * @param opClass the concrete {@link Operation Operation&lt;InputType, OutputType&gt;} implementation
351             * @param input the {@code ConcreteInputType} instance used to fill in the operation's parameters
352             * @return the resulting {@code OutputType} instance or {@code null}
353             */
354            public <
355                    InputType,
356                    ConcreteInputType extends InputType,
357                    OutputType,
358                    OpClassType extends Operation<InputType, OutputType> & ReturnsSingleRow
359            >
360            OutputType get(
361                    final Class<OpClassType> opClass,
362                    final ConcreteInputType input
363            ) {
364                    return iterate(opClass, input, new IteratorProcessor<OutputType, OutputType>() {
365                            public OutputType processIterator(Iterator<OutputType> iterator) {
366                                    if (!iterator.hasNext())
367                                            return null;
368                                    
369                                    OutputType result = iterator.next();
370                                    if (iterator.hasNext()) {
371                                            throw new IncorrectResultSizeDataAccessException(
372                                                    "Statement " + opClass.getName() + " " +
373                                                    "was declared as " + ReturnsSingleRow.class.getSimpleName() + " " +
374                                                    "but returned more than one row",
375                                                    1,
376                                                    -1 /* Actual size unknown */
377                                            );
378                                    }
379                                    
380                                    return result;
381                            }
382                            
383                    });
384            }

421            /**
422             * Executes a single-key-returning database operation
423             * (flagged as such by implementing the interfaces
424             * {@link ReturnsSingleRow}) and {@link GeneratesKey}
425             * while requires no input, and returns a single {@code KeyType}
426             * instance representing the generated key. The
427             * operation's {@code OutputType} must be {@link Void}.
428             * <p>
429             * It is an error for an operation executed with
430             * this method to return more than one key. Doing
431             * so triggers an {@link InvalidDataAccessApiUsageException}
432             * <p>
433             * If the database operation requires input (meaning
434             * {@code <InputType} is not {@link Void}) you must use
435             * {@link #key(Class, Object) key(opClass, input)}
436             * instead.
437             * 
438             * @see ReturnsSingleRow
439             * @see GeneratesKey
440             * @see #key(Class opClass)
441             * 
442             * @param <InputType> the {@link Operation Operation&lt;InputType, OutputType&gt;}'s input type
443             * @param <ConcreteInputType> the actual (concrete) implementation of {@code InputType}
444             * @param <OpClassType> the operation class'es actual type, extends {@link Operation Operation&lt;InputType, Void&gt;}
445             * @param <KeyType> the type of the generated key
446             * @param opClass the concrete {@link Operation Operation&lt;InputType, OutputType&gt;} implementation
447             * @param input the {@code ConcreteInputType} instance used to fill in the operation's parameters
448             * @return the resulting {@code KeyType} instance
449             */     
450            public
451            <
452                    InputType,
453                    ConcreteInputType extends InputType,
454                    KeyType extends Number,
455                    OpClassType extends Operation<InputType, Void> & GeneratesKey<KeyType> & ReturnsSingleRow
456            >
457            KeyType key(
458                    final Class<OpClassType> opClass,
459                    final ConcreteInputType input
460            ) {
461                    KeyMapper<KeyType> keyMapper = KeyMapper.getInstance(opClass);
462                    KeyHolder keyHolder = new GeneratedKeyHolder();
463                    
464                    m_npJdbcTemplate.update(
465                            StatementMapper.getInstance(opClass).getStatement(input),
466                            InputMapper.getInstance(opClass).mapInput(this, input),
467                            keyHolder,
468                            keyMapper.getGeneratedKeyColumns()
469                    );
470                    
471                    return keyMapper.mapKey(keyHolder);
472            }
