001    package org.phlo.tuplez.operation;
002    
003    /**
004     * Marks implementation of {@link Operation} as having
005     * a SQL statement which depends on the input parameter an
006     * can therefore not be specified by an annotation.
007     * <p>
008     * Implementations of {@link Operation} must provide a
009     * {@link #getStatement(Object) getStatement(InputType input)} method.
010     * <p>
011     * The type passed as InputType here must be assignable from
012     * instances of the database operation's InputType.
013     * 
014     * @see #getStatement(Object)
015     *
016     * @param <InputType> Interface representing a database operation's input parameters
017     */
018    public interface StatementIsComputed<InputType> extends OperationInput<InputType> {
019            /**
020             * Must return the statement to be used to execute an 
021             * {@link StatementIsComputed} implementation
022             * <p>
023             * The returned statement can use parameter references to
024             * reference input values. The only substitutions which
025             * are supposed to be done in this method are ones for which
026             * normal parameter substitution is not sufficient. One
027             * could, for example, add different "ORDER BY" clauses to
028             * the statement depending on some input value.
029             * <p>
030             * @param input the operation's input values
031             * @return the statement to be sent to the database.
032             */
033            String getStatement(final InputType input);
034    }
