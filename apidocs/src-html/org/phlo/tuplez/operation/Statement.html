001    package org.phlo.tuplez.operation;
002    
003    import java.lang.annotation.ElementType;
004    import java.lang.annotation.Retention;
005    import java.lang.annotation.RetentionPolicy;
006    import java.lang.annotation.Target;
007    
008    /**
009     * Specified the underlying statement of {@link Operation}s.
010     * <p>
011     * Every implementation of {@link Operation} must either a
012     * {@link Statement} annotation, or implement {@link StatementIsComputed}
013     * and provide a {@link StatementIsComputed#getStatement(Object)}
014     * member.
015     * <p>
016     * The mapping between the {@link Operation}'s {@code OutputType}
017     * and the statement's result columns can be adjusted by adding
018     * a {@link OutputMapping} annotation listing one or more instances
019     * of {@link OutputColumn}.
020     * 
021     * @see OutputMapping
022     * @see OutputColumn
023     * @see Operation
024     */
025    @Retention(RetentionPolicy.RUNTIME)
026    @Target(ElementType.TYPE)
027    public @interface Statement {
028            String value();
029    }
