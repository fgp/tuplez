001    package org.phlo.tuplez.operation;
002    
003    import java.lang.annotation.*;
004    
005    /**
006     * Contains a list of {@link OutputColumn}s, each mapping
007     * a certain result row to an OutputType getter and a
008     * {@link java.sql.ResultSet} accessor.
009     * 
010     * Only has an effect when applied to concrete implementations
011     * of {@link Operation}
012     */
013    @Retention(RetentionPolicy.RUNTIME)
014    @Target(ElementType.TYPE)
015    public @interface OutputMapping {
016            OutputColumn[] value();
017    }
