001    package org.phlo.tuplez.operation;
002    
003    /**
004     * Marks implementation of {@link Operation} as
005     * generating key values of type {@literal KeyType}.
006     * Usually used on operations which represent INSERT
007     * statements. 
008     * <p>
009     * This is required to execute such operations with
010     * {@link org.phlo.tuplez.Executor#key(Class, Object) Executor(opClass, input)}
011     * or {@link org.phlo.tuplez.Executor#key(Class opClass) Executor(opClass)},
012     * which return the generated key.
013     * <p>
014     * The column containing the generated key can be
015     * specified by annotating the concrete {@link Operation}
016     * instance with {@link KeyColumn}.
017     * <p>
018     * This interface defines no methods, it simply serves as
019     * a way of distinguishing at compile-time between
020     * {@link Operation}s which return generate keys and such
021     * with don't.
022     * 
023     * @see org.phlo.tuplez.Executor#key(Class, Object) Executor(opClass, input)
024     * @see org.phlo.tuplez.Executor#key(Class opClass) Executor(opClass)
025     * 
026     * @param <KeyType> the generated key's type. Must be one of
027     *                  {@link Short}, {@link Integer}, {@link Long}
028     *                  {@link java.math.BigInteger BigInteger},
029     *                  {@link java.math.BigDecimal BigDecimal}
030     */
031    
032    public interface GeneratesKey<KeyType extends Number> extends OperationOutput<Void> {
033            /* Just a marker, no members */
034    }
