001    package org.phlo.tuplez.operation;
002    
003    import java.lang.annotation.*;
004    
005    /**
006     * Specified the column containg the
007     * generated key. The column name is
008     * case-sensitive!
009     * <p>
010     * Useful only on concrete {@link Operation}
011     * implementations which are marked as
012     * {@link GeneratesKey}.
013     * <p>
014     * If this annotation is present, the
015     * operation is executed with
016     * {@link java.sql.Statement#executeUpdate(String, String[]) Statement#executeUpdate}(sql, new String[] {keyColumn})}
017     * <p>
018     * If this annotation isn't present,  the
019     * operation is executed with
020     * {@link java.sql.Statement#executeUpdate(String, int) Statement#executeUpdate}(sql, Statement.RETURN_GENERATED_KEYS)}
021     * <p>
022     * Depending on the JDBC driver's features, only
023     * one if these methods may work. The postgres
024     * JDBC driver, for example, fails in the second
025     * case since it treats <b>all</b> the table's
026     * columns as generated if no explicit column
027     * list is supplied.
028     */
029    @Retention(RetentionPolicy.RUNTIME)
030    @Target(ElementType.TYPE)
031    public @interface KeyColumn {
032            String value();
033    }
