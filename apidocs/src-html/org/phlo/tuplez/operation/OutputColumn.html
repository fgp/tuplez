001    package org.phlo.tuplez.operation;
002    
003    import java.lang.annotation.*;
004    
005    /**
006     * Overrides the OutputType's getter a given column is mapped to
007     * and the JDBC {@link java.sql.ResultSet} accessor used to
008     * fetch that column's values.
009     * 
010     * Used together with {@link OutputMapping}
011     */
012    @Retention(RetentionPolicy.RUNTIME)
013    @Target({})
014    public @interface OutputColumn {
015            String column();
016            String getter() default "";
017            String jdbcAccessor() default "";
018    }
