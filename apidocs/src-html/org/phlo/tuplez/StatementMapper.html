001    package org.phlo.tuplez;
002    
003    import java.lang.reflect.Modifier;
004    import java.util.concurrent.ConcurrentMap;
005    
006    import org.springframework.dao.InvalidDataAccessApiUsageException;
007    
008    import com.googlecode.gentyref.GenericTypeReflector;
009    
010    import org.phlo.tuplez.operation.*;
011    
012    /**
013     * Provides support for mapping database operations
014     * (implementation of {@link Operation})
015     * to statement strings
016     */
017    public final class StatementMapper<InputType> {
018            /**
019             * Statement mapper instance cache
020             */
021            private static final ConcurrentMap<Class<? extends Operation<?, ?>>, StatementMapper<?>> s_statementMappers = new java.util.concurrent.ConcurrentHashMap<Class<? extends Operation<?, ?>>, StatementMapper<?>>();
022    
023            /**
024             * Factory methods for StatementMapper instances
025             * 
026             * @param <InputType> InputType of the {@link Operation}
027             * @param opClass The concrete {@link Operation}
028             * @return An StatementMapper instance
029             */
030            public static <InputType>
031            StatementMapper<InputType>
032            getInstance(final Class<? extends Operation<InputType, ?>> opClass)
033            {
034                    @SuppressWarnings("unchecked")
035                    StatementMapper<InputType> statementMapper = (StatementMapper<InputType>)s_statementMappers.get(opClass);
036                    if (statementMapper == null) {
037                            statementMapper = new StatementMapper<InputType>(opClass);
038                            s_statementMappers.put(opClass, statementMapper);
039                    }
040                    
041                    return statementMapper;
042            }
043            
044            private final StatementIsComputed<? super InputType> m_statementIsComputedInstance;
045            private final Statement m_statementAnnotation;
046            
047            /**
048             * Constructs an StatementMapper instance for a concrete {@link Operation}
049             * 
050             * @param opClass the class object of the concrete {@link Operation}
051             */
052            private StatementMapper(final Class<? extends Operation<InputType, ?>> opClass) {
053                    /* If the operation implementation implement DatabaseOperationDynamicStatement,
054                     * we use that interface's getStatement() member. Otherwise, we look for
055                     * @Statement annotations. If both fail, we complain. If both succeed, we
056                     * complain too
057                     */
058                    /* Get annotated statement (if any) */
059                    m_statementAnnotation = opClass.getAnnotation(Statement.class);
060                    
061                    /* Check if operation requested dynamic statement */
062                    if (StatementIsComputed.class.isAssignableFrom(opClass)) {
063                            /* Do the type parameter to DatabaseOperation and StatementIsComputed match?
064                             * Note that we don't require the types to be equal, we just require
065                             * the operation's input type to be assignable to whatever was specified
066                             * as InputType of StatementIsComputed.
067                             */
068                            Class<? >opInputType = (Class<?>)GenericTypeReflector.getTypeParameter(
069                                    opClass,
070                                    OperationInput.class.getTypeParameters()[0]
071                            );
072                            Class<? >statementIsComputedInputType = (Class<?>)GenericTypeReflector.getTypeParameter(
073                                    opClass,
074                                    StatementIsComputed.class.getTypeParameters()[0]
075                            );
076                            if (!statementIsComputedInputType.isAssignableFrom(opInputType))
077                                    throw new InvalidDataAccessApiUsageException(
078                                            "Database operation class " + opClass.getName() + " " +
079                                            "specifies incompatible types for the InputTypes of " +
080                                            "StatementIsComputed and DatabaseOperation"
081                                    );
082                            @SuppressWarnings("unchecked")
083                            final Class<? extends StatementIsComputed<? super InputType>> opClassStatementIsComputed =
084                                    (Class<? extends StatementIsComputed<? super InputType>>)opClass;
085                            
086                            /* Dynamic statement operation. Annotation must not be present. */
087                            if (m_statementAnnotation != null)
088                                    throw new InvalidDataAccessApiUsageException(
089                                            "Ambiguity while determining statement for " + opClass.getName() + ". " +
090                                            "Class both implements " + StatementIsComputed.class.getSimpleName() + " " +
091                                            "and carries a @" + Statement.class.getSimpleName() + " annotation."
092                                    );
093                            
094                            /* Sanity checks. These are in principle redundant, but greatly improve the quality
095                             * of the error messages
096                             */
097    
098                            /* Is the class static? */
099                            Class<?> opEnclosingClass = opClass;
100                            do {
101                                    if ((opClass.getModifiers() & Modifier.STATIC) == 0)
102                                            throw new InvalidDataAccessApiUsageException("Database operation class " + opClass.getName() + " is a non-static inner class");
103                                    opEnclosingClass = opEnclosingClass.getEnclosingClass();
104                            } while (opEnclosingClass != null);
105                            
106                            /* Is the class default constructible? */
107                            try {
108                                    opClass.getConstructor();
109                            }
110                            catch (SecurityException e1) {
111                                    throw new InvalidDataAccessApiUsageException("Database operation class " + opClass.getName() + "'s default constructor is inaccessible");
112                            }
113                            catch (NoSuchMethodException e1) {
114                                    throw new InvalidDataAccessApiUsageException("Database operation class " + opClass.getName() + " has no default constructor");
115                            }
116                            
117                            /* Create instance */
118                            try {
119                                    m_statementIsComputedInstance = opClassStatementIsComputed.newInstance();
120                            }
121                            catch (InstantiationException e) {
122                                    throw new InvalidDataAccessApiUsageException(
123                                            "Failed to instanciate dynamic statement database operation " +
124                                            opClass.getName(),
125                                            e
126                                    );
127                            }
128                            catch (IllegalAccessException e) {
129                                    throw new InvalidDataAccessApiUsageException(
130                                            "Insufficient permissions to instanciate dynamic statement database operation " +
131                                            opClass.getName(),
132                                            e
133                                    );
134                            }
135                    }
136                    else /* !StatementIsComputed.class.isAssignableFrom(opClass) */
137                    {
138                            /* Static statement operation. Annotation must be present */
139                            if (m_statementAnnotation == null) {
140                                    throw new InvalidDataAccessApiUsageException(
141                                            "Failed to determine statement for " + opClass.getName() + ". " +
142                                            "Class neither implements " + StatementIsComputed.class.getSimpleName() + " " +
143                                            "nor does it carry @" + Statement.class.getSimpleName() + " annotation."
144                                    );
145                            }
146                            
147                            m_statementIsComputedInstance = null;
148                    }
149            }
150            
151            public String getStatement(final InputType input) {
152                    if (m_statementIsComputedInstance != null) {
153                            assert m_statementAnnotation == null;
154                            return m_statementIsComputedInstance.getStatement(input);
155                    }
156                    else {
157                            return m_statementAnnotation.value();
158                    }
159            }
160    }
