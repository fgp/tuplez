023    /**
024     * Provides support for mapping the rows resulting from
025     * an execution of an {@link Operation}'s SQL
026     * statement to objects conforming to the operation's
027     * declared output type "OutputType".
028     * 
029     * ResultSetMapper instances are typically obtained
030     * by calling {@link ResultSetMapper#getInstance}, which
031     * will create one ResultMapper instance per combination
032     * of statement call and ResultSet class.
033     * 
034     * @param <OutputType> type repesenting result rows
035     */
036    public final class ResultSetMapper<OutputType> {
037            /**
038             * ResultSetMapper instances created by {@link getInstance}
039             */
040            static private final ConcurrentMap<ID<?>, ResultSetMapper<?>> s_resultSetMappers = new java.util.concurrent.ConcurrentHashMap<ID<?>, ResultSetMapper<?>>();
041            
042            /**
043             * Mapping between the {@link Class} instances representing
044             * primitive types (like int) and the corresponding {@link Class}
045             * instances representing their boxed (object) types (like Integer)
046             */
047            static private Map<Class<?>, Class<?>> s_classPrimitiveToBoxed = new java.util.HashMap<Class<?>, Class<?>>();
048            static {
049                    s_classPrimitiveToBoxed.put(Boolean.TYPE, Boolean.class);
050                    s_classPrimitiveToBoxed.put(Character.TYPE, Character.class);
051                    s_classPrimitiveToBoxed.put(Byte.TYPE, Byte.class);
052                    s_classPrimitiveToBoxed.put(Short.TYPE, Short.class);
053                    s_classPrimitiveToBoxed.put(Integer.TYPE, Integer.class);
054                    s_classPrimitiveToBoxed.put(Long.TYPE, Long.class);
055                    s_classPrimitiveToBoxed.put(Float.TYPE, Float.class);
056                    s_classPrimitiveToBoxed.put(Double.TYPE, Double.class);
057            }
058            
059            /**
060             * Mapping between the {@link Class} instances representing
061             * boxed primitive types (like Integer) and their corresponding
062             * unboxed types (like int).
063             */
064            static private Map<Class<?>, Class<?>> s_classBoxedToPrimitive = new java.util.HashMap<Class<?>, Class<?>>();
065            static {
066                    for(Map.Entry<Class<?>, Class<?>> m: s_classPrimitiveToBoxed.entrySet())
067                            s_classBoxedToPrimitive.put(m.getValue(), m.getKey());
068            }

070            /**
071             * List of JDBC accessor methods that are ignored when
072             * autodetecting the JDBC accessor based on the output
073             * type's accessor's return type.
074             */
075            static private Set<String> s_jdbcAccessorBlackList = new java.util.HashSet<String>();
076            static {
077                    s_jdbcAccessorBlackList.add("getNString");
078                    s_jdbcAccessorBlackList.add("getLength");
079            }

081            /**
082             * Encapsulates the identifying parameters of a ResultSetMapper instance.
083             */
084            static private final class ID<OutputType> {
085                    final public Class<? extends Operation<?, OutputType>> opClass;
086                    final private Class<? extends ResultSet> rsClass;
087                    
088                    ID(
089                            final Class<? extends Operation<?, OutputType>> _opClass,
090                            final Class<? extends ResultSet> _rsClass
091                    ) {
092                            opClass = _opClass;
093                            rsClass = _rsClass;
094                    }
095                            
096                    @Override
097                    public int hashCode() {
098                            return
099                            opClass.hashCode() ^
100                            rsClass.hashCode();
101                    }
102    
103                    @Override
104                    public boolean equals(Object other) {
105                            if (this == other)
106                                    return true;
107                            if (!(other instanceof ID))
108                                    return false;
109                            
110                            ID<?> otherID = (ID<?>)other;
111                            if (opClass != otherID.opClass)
112                                    return false;
113                            if (rsClass != otherID.rsClass)
114                                    return false;
115                            return true;
116                    }
117            }
118                    
119            /**
120             * Factory method for ResultSetMapper instances.
121             * 
122             * Returns a cached ResultSetMapper if one exists with the same
123             * opClass and resultSet.class. Otherwise a new ResultSetMapper
124             * is created and put into the cache.
125             * 
126             * Note that ResultSetMapper instances depend on the ResultSet's
127             * meta data! For performance reasons, the meta data is only
128             * fetched and processed during initial construction of a 
129             * ResultSetMapper. A single operation must thus always produce
130             * ResultSets with the same meta data, otherwise row extraction
131             * will fail.
132             * 
133             * @param opClass the operation's class
134             * @param resultSet the ResultSet produced by the operation's SQL statement
135             * @param <OutputType> the operation's declared output type
136             * @return a ResultSetMapper instance
137             * @throws SQLException
138             */
139            public static <OutputType> ResultSetMapper<OutputType> getInstance(
140                    Class<? extends Operation<?, OutputType>> opClass,
141                    ResultSet resultSet
142            ) throws SQLException
143            {
144                    /* ResultSet class */
145                    Class<? extends ResultSet> resultSetClass = resultSet.getClass();
146                    
147                    /* Statement class and ResultSet class uniquely identify
148                     * a specific ResultSetMapper
149                     */
150                    ID<OutputType> rsmID = new ID<OutputType>(opClass, resultSetClass);
151                    
152                    /* Return cached ResultSetMapper or create new one if
153                     * none already exists
154                     */
155                    @SuppressWarnings("unchecked")
156                    ResultSetMapper<OutputType> resultSetMapper = (ResultSetMapper<OutputType>)s_resultSetMappers.get(rsmID);
157                    if (resultSetMapper == null) {
158                            /* Get ResultSet meta data */
159                            ResultSetMetaData resultSetMetaData;
160                            try {
161                                    resultSetMetaData = resultSet.getMetaData();
162                            }
163                            catch (SQLException e) {
164                                    throw new UncategorizedSQLException("Failed to retrieve result set meta data", null, e);
165                            }
166                            
167                            /* Create new ResultSetMapper */
168                            resultSetMapper = new ResultSetMapper<OutputType>(
169                                    opClass,
170                                    resultSetClass,
171                                    resultSetMetaData
172                            );
173                    
174                            /* Add new ResultSetMapper to the Cache. */
175                            s_resultSetMappers.put(rsmID, resultSetMapper);
176                    }
177                    
178                    return resultSetMapper;
179            }

181            /**
182             * Per-Column meta data computed by the ResultSetMapper
183             */
184            static private final class ColumnMetaData {
185                    public String columnName;
186                    public Class<?> columnClass;
187                    public String getterName;
188                    public Method getterMethod;
189                    public String jdbcAccessorName;
190                    public Method jdbcAccessorMethod;
191            }
192    
193            /* Column meta data. Computed during construction */
194            final private List<ColumnMetaData> m_columns = new java.util.ArrayList<ColumnMetaData>();
195            
196            /* Used to control the order of the fields in MapProxyInvocationHandler's toString()
197             * method
198             */
199            final private Collection<String> m_getterOrder = new java.util.ArrayList<String>();
200            
201            /* Output object meta data */
202            final private Class<?> m_outputClass;
203            
204            /* Whether to use the first column's result as output */
205            final boolean m_firstColumnIsOutput;
206            
207            private ResultSetMapper(
208                    final Class<? extends Operation<?, ?>> opClass,
209                    final Class<? extends ResultSet> rsClass,
210                    final ResultSetMetaData rsMeta
211            ) throws SQLException
212            {
213                    /* Get OutputType of statement class */
214                    m_outputClass = (Class<?>)GenericTypeReflector.getTypeParameter(
215                            opClass,
216                            OperationOutput.class.getTypeParameters()[0]
217                    );
218                    if (m_outputClass == null)
219                            throw new InvalidDataAccessApiUsageException("Unable to determine statement's output class");
220                                    
221                    /* First, fetch column names */
222                    int colNr = rsMeta.getColumnCount();
223                    for(int colIdx = 1; colIdx <= colNr; ++colIdx) {
224                            /* Create meta-data object for the column. For now,
225                             * only set the name, the rest will be filled in below
226                             */
227                            ColumnMetaData colMeta = new ColumnMetaData();
228                            colMeta.columnName = rsMeta.getColumnName(colIdx);
229                            m_columns.add(colMeta);
230                    }
231                    
232                    /* Then, fill out getter and jdbcAccessor for the fields
233                     * which have a ResultMapping annotation
234                     */
235                    {
236                            OutputMapping outputInfos = opClass.getAnnotation(OutputMapping.class);
237                            if (outputInfos != null) {
238                                    for(OutputColumn outputInfo: outputInfos.value()) {
239                                            for(ColumnMetaData colMeta: m_columns) {
240                                                    /* Join by name */
241                                                    if (!colMeta.columnName.equals(outputInfo.column()))
242                                                            continue;
243                                                    
244                                                    /* Annotations don't allow null values, thus
245                                                     * treat empty strings as null
246                                                     */
247                                                    
248                                                    if (!outputInfo.getter().isEmpty())
249                                                            colMeta.getterName = outputInfo.getter();
250            
251                                                    if (!outputInfo.jdbcAccessor().isEmpty())
252                                                            colMeta.jdbcAccessorName = outputInfo.jdbcAccessor();
253                                            }
254                                                    
255                                    }
256                            }
257                    }
258                    
259                    /* Resolve OutputType getter methods for the individual columns.
260                     * If the column wasn't explicitly mapped to a certain getter
261                     * by an annotation (parsed above), its mapped by name. Once
262                     * the getter method is resolved, the column's mapped type can
263                     * be inferred from the getter's result type.
264                     * 
265                     * For single-column result sets where the user specified
266                     * no explicit OutputType getter by an annotation, we also
267                     * check if there is a ResultSet accessor which returns a value
268                     * compatible with the whole OutputType. If there is, we use that
269                     * directly.
270                     */
271                    resolve_getters:
272                    {
273                            /* Check if we should try the single-column mode */
274                            if ((m_columns.size() == 1) &&
275                                    (m_columns.get(0).getterName == null)
276                            ) {
277                                    ColumnMetaData colMeta = m_columns.get(0);
278                                    
279                                    /* Single-column result and no explicit getter specification.
280                                     * See if we can obtain an instance of OutputType from the
281                                     * JDBC result set directly.
282                                     */
283                                    assert colMeta.columnClass == null;
284                                    colMeta.columnClass = m_outputClass;
285                                    if (computeJdbcAccessor(colMeta, rsClass)) {
286                                            /* Success */
287                                            m_firstColumnIsOutput = true;
288                                            break resolve_getters;
289                                    }
290                                    else {
291                                            /* Failure */
292                                            colMeta.columnClass = null;
293                                    }
294                            }
295                            
296                            /* Single-column mode didn't work out */
297                            m_firstColumnIsOutput = false;
298                            for(ColumnMetaData colMeta: m_columns) {
299                                    if (!computeOutputGetter(colMeta))
300                                            throw new DataRetrievalFailureException("No getter for " + colMeta.columnName + " found in " + m_outputClass.getName());
301                            }
302                            
303                            /* Store getter order */
304                            for(ColumnMetaData colMeta: m_columns)
305                                    m_getterOrder.add(colMeta.getterName);
306                    }
307                    
308                    /* Find the JDBC ResultSet accessor method to use.*/
309                    {
310                            for(ColumnMetaData colMeta: m_columns) {
311                                    if (!computeJdbcAccessor(colMeta, rsClass))
312                                            throw new DataRetrievalFailureException("No JDBC accessor for column " + colMeta.columnName + " with type " + colMeta.columnClass.getName() + " found");
313                            }
314                    }
315            }
316            
317            static final Pattern s_patternNonAlphaNumeric = Pattern.compile("[^A-Za-z0-9]*");
318            private boolean computeOutputGetter(final ColumnMetaData colMeta) {
319                    /* Skip columns who are already mapped to an OutputType getter */
320                    if (colMeta.getterMethod != null)
321                            assert colMeta.jdbcAccessorName != null;
322                    
323                    if (colMeta.getterName != null) {
324                            /* The getter's name was specified, we just need to lookup the method */
325                            try {
326                                    colMeta.getterMethod = m_outputClass.getMethod(colMeta.getterName);
327                            }
328                            catch (NoSuchMethodException e) {
329                                    throw new InvalidDataAccessApiUsageException("No getter named " + colMeta.getterName + " for column " + colMeta.columnName + " in" + m_outputClass.getName(), e);
330                            }
331                    }
332                    else {
333                            /* No getter name was specified. We try to auto-detect based on
334                             * the getter name. Since method names in java are often
335                             * CamelCased, while database field names are often wholly in
336                             * lower or upper case and user underscores as separator, we
337                             * convert the names to lowercase and remove all non-alphanumeric
338                             * characters before comparing.
339                             */
340            
341                            String getterNameComparableColumnDerived = 
342                                    "get" +
343                                    s_patternNonAlphaNumeric
344                                            .matcher(colMeta.columnName.toLowerCase(Locale.ENGLISH))
345                                            .replaceAll("");
346                            
347                            Set<Method> candidates = new java.util.HashSet<Method>();
348                            for(Method candidate: m_outputClass.getMethods()) {
349                                    String getterNameComparableCandidateDerived = s_patternNonAlphaNumeric
350                                                    .matcher(candidate.getName().toLowerCase(Locale.ENGLISH))
351                                                    .replaceAll("");
352                                    if (!getterNameComparableColumnDerived.equals(getterNameComparableCandidateDerived))
353                                            continue;
354                                    
355                                    /* Names match */
356                                    
357                                    candidates.add(candidate);
358                            }
359                            
360                            /* Check auto-detection result. If we found exactly one candidate,
361                             * we're good. We complain if the result was ambiguous, but return
362                             * false if no getter was found
363                             */
364                            
365                            switch (candidates.size()) {
366                                    case 0:
367                                            break;
368                    
369                                    case 1:
370                                            colMeta.getterMethod = candidates.iterator().next();
371                                            break;
372                    
373                                    default:
374                                            StringBuilder b = (new StringBuilder())
375                                                    .append("Ambiguous getter for column ")
376                                                    .append(colMeta.columnName)
377                                                    .append(" in ")
378                                                    .append(m_outputClass.getName())
379                                                    .append("The candidates are |");
380                                            for(Method c: candidates) {
381                                                    b.append(c.toString());
382                                                    b.append("|");
383                                            }
384                                            throw new DataRetrievalFailureException(b.toString());
385                            }
386                    }
387                    
388                    /* Update meta data and report success or failure */
389                    if (colMeta.getterMethod != null) {
390                            colMeta.getterName = colMeta.getterMethod.getName();
391                            colMeta.columnClass = colMeta.getterMethod.getReturnType();
392                            return true;
393                    }
394                    else {
395                            return false;
396                    }
397            }

399            /**
400             * Resolve the JDBC ResultSet accessor method.
401             * <p>
402             * If a jdbAccessorName is already set (usually because
403             * it was explicitly specified by an annotation), the
404             * corresponding method is simply looked up.
405             * <p>
406             * Otherwise, the accessor method is auto-detected by
407             * comparing the ResultSet's get* method's return type
408             * with the column type (which must already haven been
409             * determined)
410             *
411             * @param colMeta column meta data
412             * @param rsClass result set class
413             */
414            private boolean computeJdbcAccessor(final ColumnMetaData colMeta, final Class<? extends ResultSet> rsClass) {
415                    /* Skip columns who already are mapped to a ResultSet accessor method */
416                    if (colMeta.jdbcAccessorMethod != null) {
417                            assert colMeta.jdbcAccessorName != null;
418                            return true;
419                    }
420                    
421                    if (colMeta.jdbcAccessorName != null) {
422                            /* The accessor's name was specified, we just need to lookup the method */
423                            try {
424                                    colMeta.jdbcAccessorMethod = rsClass.getMethod(colMeta.jdbcAccessorName, new Class<?>[] {Integer.TYPE});
425                            }
426                            catch (NoSuchMethodException e) {
427                                    throw new InvalidDataAccessApiUsageException("JDBC Accessor " + colMeta.jdbcAccessorName + " for column " + colMeta.columnName + " does not exist in " + rsClass.getName(), e);
428                            }
429                            return true;
430                    }
431    
432                    /* No accessor name was specified. We try auto-detection based
433                     * on the column's type. To resolve ambiguities, we distinguish
434                     * four matching levels.
435                     *   0: Method from ResultSet interface, return type equals column type
436                     *   1: Method from ResultSet interface, return type assignable to column type
437                     *   2: Method from concrete ResultSet implementation, return type equals column type
438                     *   3: Method from concrete ResultSet implementation, return type equals column type
439                     * 
440                     * We complain if either no candidates are found, or if the first non-empty
441                     * category contains multiple accessors
442                     */
443    
444                    List<Set<Method>> candidateLevels = new java.util.ArrayList<Set<Method>>();
445                    candidateLevels.add(new java.util.HashSet<Method>());
446                    candidateLevels.add(new java.util.HashSet<Method>());
447                    candidateLevels.add(new java.util.HashSet<Method>());
448                    candidateLevels.add(new java.util.HashSet<Method>());
449                    for(Method candidate: rsClass.getMethods()) {
450                            String candidateName = candidate.getName();
451                            /* We're only interested in column value accessors,
452                             * not other methods of ResultSet. We thus limit
453                             * the search to methods named get* and which take
454                             * exactly one parameter of type "int". We also
455                             * skip candidates listed in the JDBC accessor
456                             * black list
457                             */
458                            
459                            if (!candidateName.substring(0,3).equals("get") ||
460                                s_jdbcAccessorBlackList.contains(candidateName)
461                            ) {
462                                    continue;
463                            }
464                            
465                            Class<?>[] candidateParameters = candidate.getParameterTypes();
466                            if (!(candidateParameters.length == 1) || !(candidateParameters[0].isAssignableFrom(Integer.TYPE)))
467                                    continue;
468                            
469                            /* The accessor's return type must be convertible to the
470                             * column's declared type.
471                             * 
472                             * NOTE: The column's declared class is never a primitive type,
473                             * since those cannot represent null values. But the ResultSet
474                             * accessors (like getInt()) do return primitive types (
475                             * as opposed to their boxed equivalents, e.g. getInt() returns
476                             * int, not Integer). We must thus map one to the other to
477                             * be able to detect ResultSet accessors for primitive types
478                             */
479                            Class<?> candidateClass = candidate.getReturnType();
480                            if (candidateClass.isPrimitive()) {
481                                    if (!s_classPrimitiveToBoxed.containsKey(candidateClass))
482                                            throw new DataRetrievalFailureException("Primitive type " + candidateClass.getName() + " is not supported. NOTE: This probably means that ResultSetMapper needs to be tought about this type!");
483                                    candidateClass = s_classPrimitiveToBoxed.get(candidateClass);
484                            }
485                            if (!colMeta.columnClass.isAssignableFrom(candidateClass))
486                                    continue;
487                            
488                            /* Found a candidate!
489                             * 
490                             * Now compute it's matching level
491                             */
492                            int level;
493                            
494                            /* Is it declared by the ResultSet interface? */
495                            level = 3;
496                            try {
497                                    if (ResultSet.class.getMethod(candidateName, Integer.TYPE).getReturnType().equals(candidateClass))
498                                            level = 0;
499                            }
500                            catch (NoSuchMethodException e) {
501                                    /* Ignore */
502                            }
503                            
504                            /* Does the accessor's return type match the column type exactly? */
505                            if (!candidateClass.equals(colMeta.columnClass))
506                                    level += 1;
507                            
508                            /* Store */
509                            candidateLevels.get(level).add(candidate);
510                    }
511    
512                    /* Extract accessor from candidates per the rules
513                     * outlined above
514                     */
515                    extract_candidate:
516                    for(Set<Method> candidates: candidateLevels) {
517                            switch (candidates.size()) {
518                                    case 0:
519                                            continue extract_candidate;
520    
521                                    case 1:
522                                            colMeta.jdbcAccessorMethod = candidates.iterator().next();
523                                            break extract_candidate;
524    
525                                    default:
526                                            StringBuilder b = (new StringBuilder())
527                                                    .append("Ambigous JDBC accessors for column ")
528                                                    .append(colMeta.columnName)
529                                                    .append(" with type ")
530                                                    .append(colMeta.columnClass)
531                                                    .append("The candidates are |");
532                                            for(Method c: candidates) {
533                                                    b.append(c.toString());
534                                                    b.append("|");
535                                            }
536                                            throw new DataRetrievalFailureException(b.toString());
537                            }
538                    }
539                    
540                    /* Update meta data and report success or failure */
541                    if (colMeta.jdbcAccessorMethod != null) {
542                            colMeta.jdbcAccessorName = colMeta.jdbcAccessorMethod.getName();
543                            return true;
544                    }
545                    else {
546                            return false;
547                    }
548            }

550            public OutputType mapCurrentRow(ResultSet resultSet) throws SQLException {
551                    final Map<String, Object> values = new java.util.HashMap<String, Object>();
552                    
553                    /* Extract field values from ResultSet and place into <value>,
554                     * indexed by the output class's getters names
555                     */
556                    int colIdx = 0;
557                    for(ColumnMetaData colMeta: m_columns) {
558                            ++colIdx;
559                            
560                            /* Fetch column value using the ResultSet accessor we
561                             * so painstakingly figured out
562                             */
563                            Object colValue;
564                            try {
565                                    colValue = colMeta.jdbcAccessorMethod.invoke(resultSet, colIdx);                
566                            }
567                            catch (IllegalAccessException e) {
568                                    throw new InvalidDataAccessApiUsageException("Insufficient permissions to fetch the value of column " + colMeta.columnName + " using " + colMeta.jdbcAccessorMethod.toString(), e);
569                            }
570                            catch (InvocationTargetException e) {
571                                    if (e.getTargetException() instanceof SQLException)
572                                            throw (SQLException)e.getTargetException();
573                                    else
574                                            throw new InvalidDataAccessApiUsageException("Unable to fetch the value of column " + colMeta.columnName + " using " + colMeta.jdbcAccessorMethod.toString(), e);
575                            }
576                            
577                            /* Since the accessor for primitive types return unboxed values,
578                             * their return value cannot represent null. We must thus use
579                             * wasNull() which returns true if the last value accessed by some
580                             * column accessor actually was null.
581                             */
582                            if (resultSet.wasNull())
583                                    colValue = null;
584                            
585                            /* In first-column output mode, we're done */
586                            if (m_firstColumnIsOutput) {
587                                    assert colIdx == 1;
588                                    @SuppressWarnings("unchecked")
589                                    OutputType output = (OutputType)colValue;
590                                    return output;
591                            }
592                            
593                            /* Finally store the value, keyed by the getter's name for the proxy's
594                             * convenience
595                             */
596                            values.put(colMeta.getterName, colValue);
597                    }
598                    
599                    /* Obviously not in first-column output mode */
600                    assert !m_firstColumnIsOutput;
601                    
602                    /* Creates a proxy to turn the hash into an instance of the output class */
603                    @SuppressWarnings("unchecked")
604                    OutputType output = (OutputType)Proxy.newProxyInstance(
605                            m_outputClass.getClassLoader(),
606                            new Class<?>[] {m_outputClass},
607                            new OutputProxyInvocationHandler(values, m_getterOrder)
608                    );
609                    return output;
610            }
