001    package org.phlo.tuplez;
002    
003    import java.math.BigDecimal;
004    import java.math.BigInteger;
005    import java.util.concurrent.ConcurrentMap;
006    
007    import org.springframework.dao.InvalidDataAccessApiUsageException;
008    import org.springframework.jdbc.support.KeyHolder;
009    
010    import com.googlecode.gentyref.GenericTypeReflector;
011    
012    import org.phlo.tuplez.operation.*;
013    
014    public class KeyMapper<KeyType extends Number> {
015            /**
016             * Statement mapper instance cache
017             */
018            private static final ConcurrentMap<Class<? extends Operation<?, Void>>, KeyMapper<? extends Number>> s_keyMappers = new java.util.concurrent.ConcurrentHashMap<Class<? extends Operation<?, Void>>, KeyMapper<? extends Number>>();
019    
020            /**
021             * Factory methods for StatementMapper instances
022             * 
023             * @param <KeyType> KeyType of the {@link Operation} which {@link GeneratesKey}
024             * @param opClass The concrete {@link Operation}
025             * @return An StatementMapper instance
026             */
027            public static <
028                    KeyType extends Number,
029                    OpClassType extends Operation<?, Void> & GeneratesKey<KeyType>
030            >
031            KeyMapper<KeyType>
032            getInstance(final Class<OpClassType> opClass)
033            {
034                    @SuppressWarnings("unchecked")
035                    KeyMapper<KeyType> keyMapper = (KeyMapper<KeyType>)s_keyMappers.get(opClass);
036                    if (keyMapper == null) {
037                            keyMapper = new KeyMapper<KeyType>(opClass);
038                            s_keyMappers.put(opClass, keyMapper);
039                    }
040                    
041                    return keyMapper;
042            }
043            
044            private static interface KeyExtractor<KeyType extends Number> {
045                    KeyType extractKey(KeyHolder keyHolder);
046            }
047            
048            private final KeyColumn m_generatedKeyAnnotation;
049            private final KeyExtractor<? extends Number> m_keyExtractor;
050            
051            public <
052                    OpClassType extends Operation<?, Void> & GeneratesKey<KeyType>
053            >
054            KeyMapper(final Class<OpClassType> opClass)
055            {
056                    /* Get generated key columns */
057                    m_generatedKeyAnnotation = opClass.getAnnotation(KeyColumn.class);
058    
059                    /* Get declared key type */
060                    @SuppressWarnings("unchecked")
061                    Class<KeyType> keyType = (Class<KeyType>)GenericTypeReflector.getTypeParameter(
062                            opClass,
063                            GeneratesKey.class.getTypeParameters()[0]
064                    );
065                    
066                    /* Create extractor for the declared key type */
067                    
068                    if (Short.class.isAssignableFrom(keyType)) {
069                            m_keyExtractor = new KeyExtractor<Short>() {
070                                    @Override
071                                    public Short extractKey(KeyHolder keyHolder) {
072                                            return keyHolder.getKey().shortValue();
073                                    }
074                            };
075                    }
076                    else if (Integer.class.isAssignableFrom(keyType)) {
077                            m_keyExtractor = new KeyExtractor<Integer>() {
078                                    @Override
079                                    public Integer extractKey(KeyHolder keyHolder) {
080                                            return keyHolder.getKey().intValue();
081                                    }
082                            };
083                    }
084                    else if (Long.class.isAssignableFrom(keyType)) {
085                            m_keyExtractor = new KeyExtractor<Long>() {
086                                    @Override
087                                    public Long extractKey(KeyHolder keyHolder) {
088                                            return keyHolder.getKey().longValue();
089                                    }
090                            };
091                    }
092                    else if (BigInteger.class.isAssignableFrom(keyType)) {
093                            m_keyExtractor = new KeyExtractor<BigInteger>() {
094                                    @Override
095                                    public BigInteger extractKey(KeyHolder keyHolder) {
096                                            Number key = keyHolder.getKey();
097                                            if (key instanceof BigInteger)
098                                                    return (BigInteger)key;
099                                            else
100                                                    throw new InvalidDataAccessApiUsageException(
101                                                            "Generated key is declared to be a BigInteger " +
102                                                            "but it's actually an " + key.getClass().getName()
103                                                    );
104                                    }
105                            };
106                    }
107                    else if (BigDecimal.class.isAssignableFrom(keyType)) {
108                            m_keyExtractor = new KeyExtractor<BigDecimal>() {
109                                    @Override
110                                    public BigDecimal extractKey(KeyHolder keyHolder) {
111                                            Number key = keyHolder.getKey();
112                                            if (key instanceof BigDecimal)
113                                                    return (BigDecimal)key;
114                                            else
115                                                    throw new InvalidDataAccessApiUsageException(
116                                                            "Generated key is declared to be a BigDecimal " +
117                                                            "but it's actually an " + key.getClass().getName()
118                                                    );
119                                    }
120                            };
121                    }
122                    else
123                            throw new InvalidDataAccessApiUsageException(
124                                    "Generated key type " + keyType.getName() + " is not supported"
125                            );
126            }
127            
128            public String[] getGeneratedKeyColumns() {
129                    if (m_generatedKeyAnnotation == null)
130                            return null;
131                    else
132                            return new String[] { m_generatedKeyAnnotation.value() };
133            }
134            
135            @SuppressWarnings("unchecked")
136            public KeyType mapKey(KeyHolder keyHolder) {
137                    return (KeyType)m_keyExtractor.extractKey(keyHolder);
138            }
139    }
