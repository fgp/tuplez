001    package org.phlo.tuplez;
002    
003    import java.util.Iterator;
004    
005    import org.springframework.jdbc.core.ResultSetExtractor;
006    
007    /**
008     * Processor for result iterators produced by iterate().
009     * 
010     * Required because
011     * {@link org.springframework.jdbc.core.JdbcTemplate}
012     * and
013     * {@link org.springframework.jdbc.core.namedparam.NamedParameterJdbcTemplate}
014     * close() the ResultSet after the {@link ResultSetExtractor} returns.
015     */
016    public interface IteratorProcessor<OutputType, ResultType> {
017            public ResultType processIterator(Iterator<OutputType> iterator);
018    }
